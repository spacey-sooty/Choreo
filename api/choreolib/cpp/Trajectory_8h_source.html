224 std::optional<Trajectory<SampleType>> GetSplit(int splitIndex) const {
225 if (splitIndex < 0 || splitIndex >= splits.size()) {
226 return std::nullopt;
227 }
228
229 int start = splits[splitIndex];
230 int end = (splitIndex + 1 < splits.size()) ? splits[splitIndex + 1] + 1
231 : samples.size();
232
233 auto sublist =
234 std::vector<SampleType>(samples.begin() + start, samples.begin() + end);
235 units::second_t startTime = sublist.front().GetTimestamp();
236 units::second_t endTime = sublist.back().GetTimestamp();
237
238 auto offsetSamples =
239 sublist | std::views::transform([startTime](const SampleType& s) {
240 return s.OffsetBy(-startTime);
241 });
242
243 auto filteredEvents =
244 events | std::views::filter([startTime, endTime](const auto& e) {
245 return e.timestamp >= startTime && e.timestamp <= endTime;
246 }) |
247 std::views::transform(
248 [startTime](const auto& e) { return e.OffsetBy(-startTime); });
249
250 return Trajectory<SampleType>{
251 name + "[" + std::to_string(splitIndex) + "]",
252 std::vector<SampleType>(offsetSamples.begin(), offsetSamples.end()),
253 {},
254 std::vector<EventMarker>(filteredEvents.begin(), filteredEvents.end())};
255 }

302 std::optional<SampleType> SampleInternal(units::second_t timestamp) const {
303 if (timestamp < samples[0].GetTimestamp()) {
304 return GetInitialSample();
305 }
306 if (timestamp >= GetTotalTime()) {
307 return GetFinalSample();
308 }
309
310 int low = 0;
311 int high = samples.size() - 1;
312
313 while (low != high) {
314 int mid = (low + high) / 2;
315 if (samples[mid].GetTimestamp() < timestamp) {
316 low = mid + 1;
317 } else {
318 high = mid;
319 }
320 }
321
322 if (low == 0) {
323 return samples[low];
324 }
325
326 SampleType behindState = samples[low - 1];
327 SampleType aheadState = samples[low];
328
329 if ((aheadState.GetTimestamp() - behindState.GetTimestamp()) < 1e-6_s) {
330 return aheadState;
331 }
332
333 return behindState.Interpolate(aheadState, timestamp);
334 }
