132 frc2::CommandPtr Cmd() {
133 if (trajectory.samples.size() == 0) {

244 frc2::Trigger AtTime(units::second_t timeSinceStart) {
245 if (timeSinceStart < 0_s) {
246 FRC_ReportError(frc::warn::Warning,
247 "Trigger time cannot be negative for {}", name);
248 return offTrigger;
249 }
250
251 if (timeSinceStart > trajectory.GetTotalTime()) {
252 FRC_ReportError(
253 frc::warn::Warning,
254 "Trigger time cannout be greater than total trajectory time for {}",
255 name);
256 return offTrigger;
257 }
258
259 return frc2::Trigger(loop,
260 [this, timeSinceStart, triggered = false]() mutable {
261 if (!isActive) {
262 return false;
263 }
264 if (triggered) {
265 return false;
266 }
267 if (TimeIntoTraj() >= timeSinceStart) {
268 triggered = true;
269 return true;
270 }
271 return false;
272 });
273 }

286 frc2::Trigger AtTime(std::string_view eventName) {
287 bool foundEvent = false;
288 frc2::Trigger trig = offTrigger;
289
290 for (const auto& event : trajectory.GetEvents(eventName)) {
291 trig = frc2::Trigger{trig || AtTime(event.timestamp)};
292 foundEvent = true;
293 }
294
295 if (!foundEvent) {
296 FRC_ReportError(frc::warn::Warning, "Event \"{}\" not found for {}",
297 eventName, name);
298 }
299
300 return trig;
301 }

315 frc2::Trigger AtPose(std::string_view eventName,
316 units::meter_t tolerance = DEFAULT_TOLERANCE) {
317 bool foundEvent = false;
318 frc2::Trigger trig = offTrigger;
319
320 for (const auto& event : trajectory.GetEvents(eventName)) {
321 frc::Pose2d pose =
322 trajectory
323 .template SampleAt<Year>(event.timestamp, mirrorTrajectory())
324 .GetPose();
325 trig = frc2::Trigger(trig || AtPose(pose, tolerance));
326 foundEvent = true;
327 }
328
329 if (!foundEvent) {
330 FRC_ReportError(frc::warn::Warning, "Event \"{}\" not found for {}",
331 eventName, name);
332 }
333
334 return trig;
335 }

365 void CmdInitialize() {
366 timer.Restart();
367 isActive = true;
368 for (auto& event : scheduledEvents) {
369 event.hasTriggered = false;

371 LogTrajectory(true);

374 void CmdExecute() {
375 auto sampleOpt =
376 trajectory.template SampleAt<Year>(timer.Get(), mirrorTrajectory());
377 controller(poseSupplier(), sampleOpt.value());
378 currentSample = sampleOpt.value();
379 }
380
381 void CmdEnd(bool interrupted) {
382 timer.Stop();
383 if (interrupted) {
384 controller(currentSample.GetPose(), currentSample);
385 } else {
386 controller(poseSupplier(), trajectory.GetFinalSample().value());
387 }
388 isActive = false;
389 LogTrajectory(false);
390 }

392 bool CmdIsFinished() { return timer.Get() > trajectory.GetTotalTime(); }

394 frc2::Trigger AtPose(frc::Pose2d pose, units::meter_t tolerance) {
395 frc::Translation2d checkedTrans =
396 mirrorTrajectory()
397 ? frc::Translation2d{util::fieldLength - pose.Translation().X(),
398 pose.Translation().Y()}
399 : pose.Translation();
400 return frc2::Trigger{
401 loop, [this, checkedTrans, tolerance] {
402 frc::Translation2d currentTrans = poseSupplier().Translation();
403 return currentTrans.Distance(checkedTrans) < tolerance;
404 }};
405 }

407 void AddScheduledEvent(std::string_view eventName,
408 std::function<frc2::CommandPtr()> cmdFactory) {
409 for (const auto& event : trajectory.GetEvents(eventName)) {
410 scheduledEvents.push_back({event.timestamp, std::string(eventName),
411 cmdFactory, frc2::cmd::None(), false});
412 }
413 }
414
415 void CheckAndTriggerEvents() {
416 auto currentTime = TimeIntoTraj();
417 for (auto& event : scheduledEvents) {
418 if (!event.hasTriggered && isActive && currentTime >= event.triggerTime) {
419 event.hasTriggered = true;
420 event.command = event.commandFactory();
421 event.command.Schedule();
422 }
423 }
424 }
