162template <typename T, typename U>
163 requires std::same_as<T, slp::Variable> || std::same_as<U, slp::Variable>
164slp::EqualityConstraints operator==(const Rotation2<T>& lhs,
165 const Rotation2<U>& rhs) {
166 std::vector<slp::EqualityConstraints> constraints;
167
168 // Constrain angle equality on manifold.
169 //
170 // Let lhs = <cos(a), sin(a)>. NOLINT
171 // Let rhs = <cos(b), sin(b)>. NOLINT
172 //
173 // If the angles are equal, the angle between the unit vectors should be
174 // zero.
175 //
176 // lhs x rhs = ‖lhs‖ ‖rhs‖ sin(angleBetween) NOLINT
177 // = 1 * 1 * 0
178 // = 0
179 //
180 // NOTE: angleBetween = π rad would be another solution
181 constraints.emplace_back(lhs.cos() * rhs.sin() - lhs.sin() * rhs.cos() ==
182 0.0);
183
184 // Require that lhs and rhs are unit vectors if they contain autodiff
185 // variables, since their values can change.
186 if constexpr (std::same_as<T, slp::Variable>) {
187 constraints.emplace_back(lhs.cos() * lhs.cos() + lhs.sin() * lhs.sin() ==
188 1.0);
189 }
190 if constexpr (std::same_as<U, slp::Variable>) {
191 constraints.emplace_back(rhs.cos() * rhs.cos() + rhs.sin() * rhs.sin() ==
192 1.0);
193 }
194
195 return slp::EqualityConstraints{constraints};
196}
