30 LaneConstraint(Translation2d center_line_start, Translation2d center_line_end,
31 double tolerance)
32 : m_top_line{[&] {
33 if (tolerance != 0.0) {
34 double dx = center_line_end.x() - center_line_start.x();
35 double dy = center_line_end.y() - center_line_start.y();
36 double dist = std::hypot(dx, dy);
37 auto offset = Translation2d{0.0, tolerance}.rotate_by(
38 Rotation2d{dx / dist, dy / dist});
39
40 return PointLineRegionConstraint{{0.0, 0.0},
41 center_line_start + offset,
42 center_line_end + offset,
43 Side::BELOW};
44 } else {
45 return PointLineRegionConstraint{
46 {0.0, 0.0}, center_line_start, center_line_end, Side::ON};
47 }
48 }()},
49 m_bottom_line{[&]() -> std::optional<PointLineRegionConstraint> {
50 if (tolerance != 0.0) {
51 double dx = center_line_end.x() - center_line_start.x();
52 double dy = center_line_end.y() - center_line_start.y();
53 double dist = std::hypot(dx, dy);
54 auto offset = Translation2d{0.0, tolerance}.rotate_by(
55 Rotation2d{dx / dist, dy / dist});
56
57 return PointLineRegionConstraint{{0.0, 0.0},
58 center_line_start - offset,
59 center_line_end - offset,
60 Side::ABOVE};
61 } else {
62 return std::nullopt;
63 }
64 }()} {}
