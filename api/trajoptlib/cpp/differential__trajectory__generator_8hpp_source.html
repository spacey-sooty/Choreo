161class TRAJOPT_DLLEXPORT DifferentialTrajectory {
162 public:
164 std::vector<DifferentialTrajectorySample> samples;
165
166 DifferentialTrajectory() = default;
167
173 explicit DifferentialTrajectory(
174 std::vector<DifferentialTrajectorySample> samples)
175 : samples{std::move(samples)} {}
176
182 explicit DifferentialTrajectory(const DifferentialSolution& solution) {
183 double ts = 0.0;
184 for (size_t sample = 0; sample < solution.x.size(); ++sample) {
185 samples.emplace_back(
186 ts, solution.x[sample], solution.y[sample], solution.heading[sample],
187 solution.vl[sample], solution.vr[sample],
188 solution.angular_velocity[sample], solution.al[sample],
189 solution.ar[sample], solution.Fl[sample], solution.Fr[sample]);
190 ts += solution.dt[sample];
191 }
192 }
193};
