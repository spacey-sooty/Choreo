125 void WptObstacle(size_t index, const Obstacle& obstacle) {
126 for (auto& _bumpers : bumpers) {
127 auto minDistance = _bumpers.safetyDistance + obstacle.safetyDistance;
128
129 size_t bumperCornerCount = _bumpers.points.size();
130 size_t obstacleCornerCount = obstacle.points.size();
131 if (bumperCornerCount == 1 && obstacleCornerCount == 1) {
132 // if the bumpers and obstacle are only one point
133 WptConstraint(index,
134 PointPointConstraint{_bumpers.points.at(0),
135 obstacle.points.at(0), minDistance});
136 return;
137 }
138
139 // robot bumper edge to obstacle point constraints
140 for (auto& obstaclePoint : obstacle.points) {
141 // First apply constraint for all but last edge
142 for (size_t bumperCornerIndex = 0;
143 bumperCornerIndex < bumperCornerCount - 1; ++bumperCornerIndex) {
144 WptConstraint(index, LinePointConstraint{
145 _bumpers.points.at(bumperCornerIndex),
146 _bumpers.points.at(bumperCornerIndex + 1),
147 obstaclePoint, minDistance});
148 }
149 // apply to last edge: the edge connecting the last point to the first
150 // must have at least three points to need this
151 if (bumperCornerCount >= 3) {
152 WptConstraint(index,
153 LinePointConstraint{
154 _bumpers.points.at(bumperCornerCount - 1),
155 _bumpers.points.at(0), obstaclePoint, minDistance});
156 }
157 }
158
159 // obstacle edge to bumper corner constraints
160 for (auto& bumperCorner : _bumpers.points) {
161 if (obstacleCornerCount > 1) {
162 for (size_t obstacleCornerIndex = 0;
163 obstacleCornerIndex < obstacleCornerCount - 1;
164 ++obstacleCornerIndex) {
165 WptConstraint(
166 index,
167 PointLineConstraint{
168 bumperCorner, obstacle.points.at(obstacleCornerIndex),
169 obstacle.points.at(obstacleCornerIndex + 1), minDistance});
170 }
171 if (obstacleCornerCount >= 3) {
172 WptConstraint(index, PointLineConstraint{
173 bumperCorner,
174 obstacle.points.at(bumperCornerCount - 1),
175 obstacle.points.at(0), minDistance});
176 }
177 } else {
178 WptConstraint(
179 index, PointPointConstraint{bumperCorner, obstacle.points.at(0),
180 minDistance});
181 }
182 }
183 }
184 }

278 void SgmtConstraint(size_t fromIndex, size_t toIndex,
279 const Constraint& constraint) {
280 assert(fromIndex < toIndex);
281
282 NewWpts(toIndex);
283 path.waypoints.at(fromIndex).waypointConstraints.push_back(constraint);
284 for (size_t index = fromIndex + 1; index <= toIndex; ++index) {
285 path.waypoints.at(index).waypointConstraints.push_back(constraint);
286 path.waypoints.at(index).segmentConstraints.push_back(constraint);
287 }

354 void NewWpts(size_t finalIndex) {
355 if (finalIndex < path.waypoints.size()) {
356 return;
357 }
358
359 for (size_t i = path.waypoints.size(); i <= finalIndex; ++i) {
360 path.waypoints.emplace_back();
361 initialGuessPoints.emplace_back(std::vector<Pose2d>{{0.0, 0.0, {0.0}}});
362 if (i != 0) {
363 controlIntervalCounts.push_back(40);
364 }
365 }
